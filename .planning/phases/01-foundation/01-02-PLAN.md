   ```typescript
   export function parseWordList(text: string): Set<string> {

       text.split('\n').map(w => w.trim().toLowerCase()).filter(Boolean)
     )
   }
   ```

   - Interface: `{ words: Set<string>, status: 'idle' | 'loading' | 'ready' | 'error', loadDictionary: () => Promise<void> }`
   - `loadDictionary`: guard with `if (get().status !== 'idle') return`, set loading, fetch `/TWL06.txt`, check `res.ok`, parse with `parseWordList`, set words + ready. Catch block sets error.
   - Export as `useDictionaryStore`
